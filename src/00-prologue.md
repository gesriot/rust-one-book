`src/main.rs`:

```rust
use std::env;
use std::fs;
use std::io::{self, Read};

struct Stats {
    lines: usize,
    words: usize,
    chars: usize,
    bytes: usize,
}

struct Options {
    lines: bool,
    words: bool,
    bytes: bool,
    chars: bool,
}

impl Options {
    fn default_all() -> Self {
        Options {
            lines: true,
            words: true,
            bytes: true,
            chars: false,
        }
    }
    
    fn none() -> Self {
        Options {
            lines: false,
            words: false,
            bytes: false,
            chars: false,
        }
    }
    
    fn any_set(&self) -> bool {
        self.lines || self.words || self.bytes || self.chars
    }
}

fn count_stats(content: &str) -> Stats {
    Stats {
        lines: content.lines().count(),
        words: content.split_whitespace().count(),
        chars: content.chars().count(),
        bytes: content.len(),
    }
}

fn read_stdin() -> io::Result<String> {
    let mut buffer = String::new();
    io::stdin().read_to_string(&mut buffer)?;
    Ok(buffer)
}

fn print_stats(stats: &Stats, opts: &Options, filename: Option<&str>) {
    if opts.lines {
        print!("{:>8}", stats.lines);
    }
    if opts.words {
        print!("{:>8}", stats.words);
    }
    if opts.bytes {
        print!("{:>8}", stats.bytes);
    }
    if opts.chars {
        print!("{:>8}", stats.chars);
    }
    if let Some(name) = filename {
        print!(" {}", name);
    }
    println!();
}

fn process_file(path: &str) -> io::Result<Stats> {
    let content = fs::read_to_string(path)?;
    Ok(count_stats(&content))
}

fn parse_args() -> (Options, Vec<String>) {
    let args: Vec<String> = env::args().skip(1).collect();
    let mut opts = Options::none();
    let mut files = Vec::new();
    
    for arg in args {
        if arg.starts_with('-') && arg.len() > 1 {
            for c in arg[1..].chars() {
                match c {
                    'l' => opts.lines = true,
                    'w' => opts.words = true,
                    'c' => opts.bytes = true,
                    'm' => opts.chars = true,
                    _ => eprintln!("wordcount: –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω—ã–π —Ñ–ª–∞–≥: -{}", c),
                }
            }
        } else {
            files.push(arg);
        }
    }
    
    if !opts.any_set() {
        opts = Options::default_all();
    }
    
    (opts, files)
}

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let (opts, files) = parse_args();
    
    if files.is_empty() {
        let content = read_stdin()?;
        let stats = count_stats(&content);
        print_stats(&stats, &opts, None);
        return Ok(());
    }
    
    let mut total = Stats {
        lines: 0,
        words: 0,
        chars: 0,
        bytes: 0,
    };
    let mut file_count = 0;
    
    for path in &files {
        match process_file(path) {
            Ok(stats) => {
                print_stats(&stats, &opts, Some(path));
                total.lines += stats.lines;
                total.words += stats.words;
                total.chars += stats.chars;
                total.bytes += stats.bytes;
                file_count += 1;
            }
            Err(e) => {
                eprintln!("wordcount: {}: {}", path, e);
            }
        }
    }
    
    if file_count > 1 {
        print_stats(&total, &opts, Some("total"));
    }
    
    Ok(())
}
```
